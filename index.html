    function escapeHtml(s) {

      return String(s ?? "").replace(/[&<>"']/g, (c) => ({

        "&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;","'":"&#039;"

      }[c]));

    }


    /***********************

     * POLICY A FILTERING

     * - Age is required and always hard-filtered

     * - Setting/language/insurance are filtered only when explicitly incompatible

     * - Unknown/missing fields are included and shown as "Verify"

     ***********************/

    function includesOrUnknown(arr, value) {

      if (!Array.isArray(arr) || arr.length === 0) return "unknown";

      return arr.includes(value) ? "yes" : "no";

    }


    function settingMatchOrUnknown(listingSetting, selectedSetting) {

      if (selectedSetting === "either") return "yes";

      if (!Array.isArray(listingSetting) || listingSetting.length === 0) return "unknown";


      const hasInPerson = listingSetting.includes("in_person") || listingSetting.includes("hybrid");

      const hasTele = listingSetting.includes("telehealth") || listingSetting.includes("hybrid");


      if (selectedSetting === "in_person") return hasInPerson ? "yes" : "no";

      if (selectedSetting === "telehealth") return hasTele ? "yes" : "no";

      return "unknown";

    }


    function matchesFilters(p, f) {

      // 1) Age is required and always hard

      if (!Array.isArray(p.ages) || p.ages.length === 0) return false;

      if (!p.ages.includes(f.ageBand)) return false;


      // 2) Setting: hard only if explicit mismatch; unknown is allowed

      const settingRes = settingMatchOrUnknown(p.setting, f.setting);

      if (settingRes === "no") return false;


      // 3) Language: if "any", ignore; else mismatch only excludes when explicitly known

      if (f.language !== "any") {

        const langRes = includesOrUnknown(p.language, f.language);

        if (langRes === "no") return false;

      }


      // 4) Insurance: if "unknown", ignore; else mismatch only excludes when explicitly known

      if (f.insurance !== "unknown") {

        const insRes = includesOrUnknown(p.insurance, f.insurance);

        if (insRes === "no") return false;

      }


      return true;

    }

    function setActiveTab(prefix, which) {

      const tabs = [

        { tab: `${prefix}TabResources`, panel: `${prefix}PanelResources`, key: "Resources" },

        { tab: `${prefix}TabWhat`, panel: `${prefix}PanelWhat`, key: "What" },

        { tab: `${prefix}TabLike`, panel: `${prefix}PanelLike`, key: "Like" },

      ];

      for (const t of tabs) {

        const tabEl = el(t.tab);

        const panelEl = el(t.panel);

        const active = (t.key === which);

        tabEl.classList.toggle("active", active);

        tabEl.setAttribute("aria-selected", active ? "true" : "false");

        panelEl.style.display = active ? "" : "none";

      }

    }

    function renderResults(filters) {

      renderChips(filters);


      const eligible = PROVIDERS.filter(p => matchesFilters(p, filters));


      const php = sortAlpha(eligible.filter(p => p.level === "PHP"));

      const iop = sortAlpha(eligible.filter(p => p.level === "IOP"));

      const th  = sortAlpha(eligible.filter(p => p.level === "THERAPY"));


      renderProviderCards(php, "phpResults");

      renderProviderCards(iop, "iopResults");

      renderProviderCards(th, "therapyResults");


      updateCounts(php.length, iop.length, th.length);


      // Default tabs: Resources

      setActiveTab("php", "Resources");

      setActiveTab("iop", "Resources");

      setActiveTab("th", "Resources");


      // Neutral convenience only: open the section matching "seeking" by scrolling

      // (does not hide or rank anything)

      const seeking = filters.seeking;

      if (seeking === "php") el("phpResults").scrollIntoView({ behavior: "smooth", block: "start" });

      if (seeking === "iop") el("iopResults").scrollIntoView({ behavior: "smooth", block: "start" });

      if (seeking === "therapy") el("therapyResults").scrollIntoView({ behavior: "smooth", block: "start" });

    }
